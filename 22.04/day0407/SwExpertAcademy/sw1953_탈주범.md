package day0407;

import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.LinkedList;
import java.util.Queue;
import java.util.StringTokenizer;

	static int N, M, R, C, L; // 세로크기 : N, 가로크기 M, 맨홀 뚜껑 위치의 세로위치 : R, 맨홀 가로위치 : C, 탈출 후 소요된 시간 L
	static int map[][];
	static boolean visited[][];
	static int di[][] = { { -1, 0 }, { 1, 0 }, { 0, -1 }, { 0, 1 } }; // 상 하 좌 우

	public static void main(String[] args) throws NumberFormatException, IOException {
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		int TC = Integer.parseInt(br.readLine());
		for (int tc = 1; tc <= TC; tc++) {
			StringTokenizer st = new StringTokenizer(br.readLine());
			N = Integer.parseInt(st.nextToken());
			M = Integer.parseInt(st.nextToken());
			R = Integer.parseInt(st.nextToken());
			C = Integer.parseInt(st.nextToken());
			L = Integer.parseInt(st.nextToken());
			map = new int[N][M];
			visited = new boolean[N][M];
			for (int i = 0; i < N; i++) {
				st = new StringTokenizer(br.readLine());
				for (int j = 0; j < M; j++) {
					map[i][j] = Integer.parseInt(st.nextToken());
				}
			}
			bfs(R, C, 1);
			
			int cnt = 0;
			for (int i = 0; i < N; i++)
				for (int j = 0; j < M; j++) {
					if (visited[i][j])
						cnt++;
				}
			System.out.println("#"+tc+" "+ cnt);
		}
	}

	static void bfs(int y, int x, int time) {
		Queue<int[]> q = new LinkedList<int[]>();
		q.add(new int[] { y, x, time });

		while (!q.isEmpty()) {
			int qy = q.peek()[0];
			int qx = q.peek()[1];
			int qtime = q.peek()[2];
			q.poll();
			visited[qy][qx] = true;

			if (qtime == L) {
				break;
			}

			switch (map[qy][qx]) {
			case 1:
				for (int i = 0; i < 4; i++) {
					int nextY = qy + di[i][0];
					int nextX = qx + di[i][1];

							|| map[nextY][nextX] == 0)
						continue;

						if (map[nextY][nextX] == 1 || map[nextY][nextX] == 2 || map[nextY][nextX] == 5
								|| map[nextY][nextX] == 6) {
							q.add(new int[] { nextY, nextX, qtime + 1 });
							visited[nextY][nextX] = true;
						}
					} else if (i == 1) {
						if (map[nextY][nextX] == 1 || map[nextY][nextX] == 2 || map[nextY][nextX] == 4
								|| map[nextY][nextX] == 7) {
							q.add(new int[] { nextY, nextX, qtime + 1 });
							visited[nextY][nextX] = true;
						}
					} else if (i == 2) {
						if (map[nextY][nextX] == 1 || map[nextY][nextX] == 3 || map[nextY][nextX] == 4
								|| map[nextY][nextX] == 5) {
							q.add(new int[] { nextY, nextX, qtime + 1 });
							visited[nextY][nextX] = true;
						}
					} else if (i == 3) {
						if (map[nextY][nextX] == 1 || map[nextY][nextX] == 3 || map[nextY][nextX] == 6
								|| map[nextY][nextX] == 7) {
							q.add(new int[] { nextY, nextX, qtime + 1 });
							visited[nextY][nextX] = true;
						}
					}
				}
				break;
			case 2:
				for (int i = 0; i < 2; i++) {
					int nextY = qy + di[i][0];
					int nextX = qx + di[i][1];
					if (nextY < 0 || nextY >= N || nextX < 0 || nextX >= M || visited[nextY][nextX]
							|| map[nextY][nextX] == 0)
						continue;
					if (i == 0) {
						if (map[nextY][nextX] == 1 || map[nextY][nextX] == 2 || map[nextY][nextX] == 5
								|| map[nextY][nextX] == 6) {
							q.add(new int[] { nextY, nextX, qtime + 1 });
							visited[nextY][nextX] = true;
						}
					} else {
						if (map[nextY][nextX] == 1 || map[nextY][nextX] == 2 || map[nextY][nextX] == 4
								|| map[nextY][nextX] == 7) {
							q.add(new int[] { nextY, nextX, qtime + 1 });
							visited[nextY][nextX] = true;
						}
					}
				}
				break;
			case 3:
				for (int i = 2; i < 4; i++) {
					int nextY = qy + di[i][0];
					int nextX = qx + di[i][1];
					if (nextY < 0 || nextY >= N || nextX < 0 || nextX >= M || visited[nextY][nextX]
							|| map[nextY][nextX] == 0)
						continue;
					if (i == 2) {
						if (map[nextY][nextX] == 1 || map[nextY][nextX] == 3 || map[nextY][nextX] == 4
								|| map[nextY][nextX] == 5) {
							q.add(new int[] { nextY, nextX, qtime + 1 });
							visited[nextY][nextX] = true;
						}
					} else {
						if (map[nextY][nextX] == 1 || map[nextY][nextX] == 3 || map[nextY][nextX] == 6
								|| map[nextY][nextX] == 7) {
							q.add(new int[] { nextY, nextX, qtime + 1 });
							visited[nextY][nextX] = true;
						}
					}
				}
				break;
			case 4:
				for (int i = 0; i < 4; i += 3) {
					int nextY = qy + di[i][0];
					int nextX = qx + di[i][1];
					if (nextY < 0 || nextY >= N || nextX < 0 || nextX >= M || visited[nextY][nextX]
							|| map[nextY][nextX] == 0)
						continue;
					if (i == 0) {
						if (map[nextY][nextX] == 1 || map[nextY][nextX] == 2 || map[nextY][nextX] == 5
								|| map[nextY][nextX] == 6) {
							q.add(new int[] { nextY, nextX, qtime + 1 });
							visited[nextY][nextX] = true;

						if (map[nextY][nextX] == 1 || map[nextY][nextX] == 3 || map[nextY][nextX] == 6
								|| map[nextY][nextX] == 7) {
							q.add(new int[] { nextY, nextX, qtime + 1 });
							visited[nextY][nextX] = true;
						}
					}
				}
				break;
			case 5:
				for (int i = 1; i < 4; i += 2) {
					int nextY = qy + di[i][0];
					int nextX = qx + di[i][1];
					if (nextY < 0 || nextY >= N || nextX < 0 || nextX >= M || visited[nextY][nextX]

							q.add(new int[] { nextY, nextX, qtime + 1 });
							visited[nextY][nextX] = true;
						}
					} else if (i == 3) {
						if (map[nextY][nextX] == 1 || map[nextY][nextX] == 3 || map[nextY][nextX] == 6
								|| map[nextY][nextX] == 7) {
							q.add(new int[] { nextY, nextX, qtime + 1 });
							visited[nextY][nextX] = true;
						}
					}
				}
				break;
			case 6:
				for (int i = 1; i < 3; i++) {
					int nextY = qy + di[i][0];
					int nextX = qx + di[i][1];
					if (nextY < 0 || nextY >= N || nextX < 0 || nextX >= M || visited[nextY][nextX]
							|| map[nextY][nextX] == 0)
						continue;
					if (i == 1) {
						if (map[nextY][nextX] == 1 || map[nextY][nextX] == 2 || map[nextY][nextX] == 4

								|| map[nextY][nextX] == 5) {

							visited[nextY][nextX] = true;
						}
					}
				}
				break;
			case 7:
				for (int i = 0; i < 3; i += 2) {
					int nextY = qy + di[i][0];
					int nextX = qx + di[i][1];
					if (nextY < 0 || nextY >= N || nextX < 0 || nextX >= M || visited[nextY][nextX]

								|| map[nextY][nextX] == 6) {

							q.add(new int[] { nextY, nextX, qtime + 1 });

							q.add(new int[] { nextY, nextX, qtime + 1 });
							visited[nextY][nextX] = true;

				break;
			}
		}
	}

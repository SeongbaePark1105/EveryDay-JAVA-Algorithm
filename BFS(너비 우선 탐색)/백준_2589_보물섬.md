# 백준_2589_보물섬

- 난이도 : 골드 5

- 외부링크 : [bk2589](https://www.acmicpc.net/problem/2589)

- 접근 방식

  > 처음에는 DFS방식으로 접근을 했는데 생각했던 알고리즘은 짰으나 값이 안나와서 무엇이 잘못된지 몰랐었다.
  >
  > 알고보니 BFS방식으로 푸는 문제였고 브루트 포스도 적용해야 하는 문제여서 브루트 포스와 BFS 알고리즘 방식으로 풀었다.

```java
import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.LinkedList;
import java.util.Queue;
import java.util.StringTokenizer;

public class bk2589_보물섬 {
	static int Y, X;
	static char[][] map;
	static boolean visited[][];
	static int di[][] = { { -1, 0 }, { 1, 0 }, { 0, -1 }, { 0, 1 } };
	static int result = Integer.MIN_VALUE;

	public static void main(String[] args) throws IOException {
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		StringTokenizer st = new StringTokenizer(br.readLine());
		Y = Integer.parseInt(st.nextToken());
		X = Integer.parseInt(st.nextToken());
		map = new char[Y][X];
		visited = new boolean[Y][X];

		for (int i = 0; i < Y; i++) {
			map[i] = br.readLine().toCharArray();
		}
		int answer = Integer.MIN_VALUE;
		for (int i = 0; i < Y; i++) {
			for (int j = 0; j < X; j++) {
				if (map[i][j] == 'L') {
					visited = new boolean[Y][X];
					bfs(i, j);
					answer = Math.max(answer, result);
				}
			}
		}
		System.out.println(answer);
	}

	static void bfs(int y, int x) {
		Queue<int[]> q = new LinkedList<>();
		q.add(new int[] { y, x, 0 });
		while (!q.isEmpty()) {
			int newY = q.peek()[0];
			int newX = q.peek()[1];
			int cnt = q.peek()[2];
			result = Math.max(result, cnt);
			visited[newY][newX] = true;
			q.poll();
			for (int i = 0; i < 4; i++) {
				int dy = newY + di[i][0];
				int dx = newX + di[i][1];
				if (dy >= 0 && dx >= 0 && dy < Y && dx < X && map[dy][dx] == 'L' && !visited[dy][dx]) {
					q.add(new int[] { dy, dx, cnt + 1 });
					visited[dy][dx] = true;
				}
			}
		}
	}
}
```

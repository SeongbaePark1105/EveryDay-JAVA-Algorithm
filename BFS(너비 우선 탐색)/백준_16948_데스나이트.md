# 백준_16948_데스나이트

- 난이도 : 실버 1

- 외부링크 : [bk16948](https://www.acmicpc.net/problem/16948)

- 접근 방식
  > BFS 방식으로 접근해서 풀었다.
  > 
  > 방문 체크를 해서 방문한 곳은 다시 방문하지 않게 해주었고
  > 
  > 먼저 방문하면 모든 while문을 끝내도록 하였다.
  >
  > 만약에 모든 곳을 방문했는데 도착하지 못하면 -1을 출력해주었다.
```java
package baekjoon;

import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.LinkedList;
import java.util.Queue;
import java.util.StringTokenizer;

public class bk16948_데스나이트 {
    static int N;
    static int r1, r2, c1, c2;
    static boolean visited[][];
    static int arr[][];
    static int di[][] = { {-2, -1}, {-2 , 1}, {0, -2}, {0, 2}, {2, -1}, {2, 1}  };
    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        N = Integer.parseInt(br.readLine());
        StringTokenizer st = new StringTokenizer(br.readLine());
        r1 = Integer.parseInt(st.nextToken());
        c1 = Integer.parseInt(st.nextToken());
        r2 = Integer.parseInt(st.nextToken());
        c2 = Integer.parseInt(st.nextToken());
        visited = new boolean[N+1][N+1];
        arr = new int[N+1][N+1];
        bfs();
    }
    static void bfs(){
        Queue<int[]> q= new LinkedList<>();
        q.add(new int[]{r1, c1, 0});
        visited[r1][c1] = true;
        while(!q.isEmpty()){
            int y = q.peek()[0];
            int x = q.peek()[1];
            int cnt = q.peek()[2];
            q.poll();
            for(int i = 0; i < 6; i++){
                int ny = y + di[i][0];
                int nx = x + di[i][1];
                if(ny < 0 || nx < 0 || ny>N || nx>N || visited[ny][nx])
                    continue;
                if(ny == r2 && nx == c2){
                    System.out.println(cnt+1);
                    return ;
                }
                visited[ny][nx] = true;
                q.add(new int[]{ny, nx, cnt+1});
            }
        }
        System.out.println(-1);
    }
}
```

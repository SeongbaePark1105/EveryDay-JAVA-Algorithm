백준 1992번 쿼드 트리
=============
* 난이도 : 실버 1
* 외부 링크 : [SW4012](https://www.acmicpc.net/problem/1992)

* 접근 방식
	+ 저번 스터디때 풀었던 1780번 종이의 개수와 비슷한 문제여서 재귀함수를 이용해서 풀었다.


```java
package baekjoon;

import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.StringTokenizer;

public class bk1992 {
	static int N;
	static int arr[][];

	public static void main(String[] args) throws NumberFormatException, IOException {
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		N = Integer.parseInt(br.readLine());
		arr = new int[N][N];
		for (int i = 0; i < N; i++) {
			String[] str = br.readLine().split("");
			for (int j = 0; j < N; j++) {
				arr[i][j] = Integer.parseInt(str[j]);
			}
		}
		Qtree(0, 0, N);

	}

	static void Qtree(int y, int x, int size) {
		if (checkQtree(y, x, size)) {
			System.out.print(arr[y][x]);
			return;
		}
		int newsize = size / 2;
		System.out.print("(");
		Qtree(y, x, newsize);
		Qtree(y, x + newsize, newsize);
		Qtree(y + newsize, x, newsize);
		Qtree(y + newsize, x + newsize, newsize);
		System.out.print(")");

	}

	static boolean checkQtree(int y, int x, int size) {
		int result = arr[y][x];
		for (int i = y; i < y + size; i++) {
			for (int j = x; j < x + size; j++) {
				if (arr[i][j] != result) {
					return false;
				}
			}
		}
		return true;
	}
}

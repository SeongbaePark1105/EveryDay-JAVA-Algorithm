# 백준 #14999 [주사위 굴리기](https://www.acmicpc.net/problem/14999)

### 난이도 `골드 4` `구현` `시뮬레아션` 

---

#### 접근 방식

> 단순한 구현입니다.

#### 풀이

```java
import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.StringTokenizer;

public class bk14499_주사위굴리기 {
	static int N, M, x, y, K;
	static int map[][];
	static int dice[] = new int[6]; 
	// 주사위 윗면 : 0 , 아랫면 : 1, 앞면(내 기준 남쪽) :2, 뒷면 (내 기준 북쪽 ) : 3, 왼쪽 : 4, 오른쪽 : 5

	public static void main(String[] args) throws IOException {
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		StringTokenizer st = new StringTokenizer(br.readLine());
		N = Integer.parseInt(st.nextToken());
		M = Integer.parseInt(st.nextToken());
		y = Integer.parseInt(st.nextToken()); // 열
		x = Integer.parseInt(st.nextToken()); // 행
		K = Integer.parseInt(st.nextToken());
		map = new int[N][M];
		for (int i = 0; i < N; i++) {
			st = new StringTokenizer(br.readLine());
			for (int j = 0; j < M; j++) {
				map[i][j] = Integer.parseInt(st.nextToken());
			}
		}
		st = new StringTokenizer(br.readLine());
		for (int i = 0; i < K; i++) {
			int dir = Integer.parseInt(st.nextToken()); // 1 : 동쪽  , 2 : 서쪽, 3: 북쪽, 4: 남쪽
			// 주사위 윗면 : 0 , 아랫면 : 1, 앞면(내 기준 남쪽) :2, 뒷면 (내 기준 북쪽 ) : 3, 왼쪽 : 4, 오른쪽 : 5
			if (dir == 1) { // 동쪽
				if(x + 1 >=M)
					continue;
				x++; // 동쪽, 즉 오른쪽으로 1칸 이동했으니 ++
				int temp = dice[1]; // 아랫면 값 저장
				dice[1] = dice[5];
				dice[5] = dice[0];
				dice[0] = dice[4];
				dice[4] = temp;
			} else if (dir == 2) { //서쪽
				if (x - 1 < 0)
					continue;
				x--;
				int temp = dice[1];
				dice[1] = dice[4];
				dice[4] = dice[0];
				dice[0] = dice[5];
				dice[5] = temp;
			} else if (dir == 3) { // 북쪽
				if (y -1 < 0)
					continue;
				y--;
				int temp = dice[1];
				dice[1] = dice[3];
				dice[3] = dice[0];
				dice[0] = dice[2];
				dice[2] = temp;
			} else if (dir ==4) {
				if (y + 1 >=N)
					continue;
				y++;
				int temp = dice[1];
				dice[1] = dice[2];
				dice[2] = dice[0];
				dice[0] = dice[3];
				dice[3] = temp;
			}
			
			if  (map[y][x] == 0) {
				map[y][x] = dice[1];
			}else if (map[y][x] != 0){
				dice[1] =  map[y][x];
				map[y][x] = 0;
			}
			System.out.println(dice[0]);
		}
	}
}
```
